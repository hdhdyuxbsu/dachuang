For Immediate Use (3 Steps):

1. Copy char_bitmap.h to your ESP32 project folder

2. Include in your C code:
   #include "char_bitmap.h"

3. Display a character:
   const uint16_t *bitmap = get_char_bitmap(0x6E29);  // 温
   for (int row = 0; row < 12; row++) {
       for (int col = 0; col < 12; col++) {
           int pixel = (bitmap[row] >> (15 - col)) & 1;
           if (pixel) {
               drawPixel(x + col, y + row, color);
           }
       }
   }
